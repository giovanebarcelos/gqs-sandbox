import static org.mockito.Mockito.*;

        Calculator mockedCalculator = mock(Calculator.class);
        when(mockedCalculator.subtract(5.0, 0.0)).thenReturn(5.0);
        assertEquals(5.0, mockedCalculator.subtract(5.0, 0.0), 0.001); 
    }

    @Test
    public void testMultiplication() {
        Calculator calculator = new Calculator();
        assertEquals(6.0, calculator.multiply(2.0, 3.0), 0.001); // Testa multiplicação de dois números positivos

        when(mockedCalculator.multiply(2.0, -5.0)).thenReturn(-10.0);
        assertEquals(-10.0, mockedCalculator.multiply(2.0, -5.0), 0.001); 
    }

    @Test
    public void testDivision() {
        Calculator calculator = new Calculator();
        assertEquals(2.0, calculator.divide(6.0, 3.0), 0.001); // Testa divisão de dois números positivos (onde o primeiro número é divisível pelo segundo)
        
        // Simulando a divisão de um número positivo por zero (deve lançar uma exceção ArithmeticException)
        Calculator mockedCalculator = mock(Calculator.class);
        doThrow(new ArithmeticException()).when(mockedCalculator).divide(5.0, 0.0);
        try {
            mockedCalculator.divide(5.0, 0.0);
            fail("Expected an ArithmeticException to be thrown");
        } catch (ArithmeticException e) {
            // Test passed
        }
    }
}

Classe Testada Utilizando o JUnit e o Mockito:

Aqui está a implementação dos testes unitários para a classe EmailService utilizando o JUnit e o Mockito:

java

import static org.junit.Assert.*;
import static org.mockito.Mockito.*;

import javax.mail.*;
import javax.mail.internet.*;

import org.junit.*;
import org.mockito.*;

public class EmailServiceTest {

    @Test
    public void testSendEmail_Success() throws MessagingException {
        // Configuração do mock
        Session session = mock(Session.class);
        Transport transport = mock(Transport.class);
        when(session.getTransport("smtp")).thenReturn(transport);

        // Configuração do serviço de email
        EmailService emailService = new EmailService(session, "username", "password");

        // Envio do email
        emailService.sendEmail("destinatario@example.com", "Assunto", "Corpo");

        // Verificação se o email foi enviado com sucesso
        verify(transport).sendMessage(any(Message.class), any(Address[].class));
    }

    @Test(expected = MessagingException.class)
    public void testSendEmail_AuthenticationFailure() throws MessagingException {
        // Configuração do mock para simular falha na autenticação
        Session session = mock(Session.class);
        when(session.getTransport("smtp")).thenThrow(new AuthenticationFailedException());

        // Configuração do serviço de email
        EmailService emailService = new EmailService(session, "username", "password");

        // Envio do email deve lançar exceção devido à falha na autenticação
        emailService.sendEmail("destinatario@example.com", "Assunto", "Corpo");
    }

    @Test(expected = MessagingException.class)
    public void testSendEmail_InvalidRecipient() throws MessagingException {
        // Configuração do mock para simular falha no envio de email devido a um destinatário inválido
        Session session = mock(Session.class);
        Transport transport = mock(Transport.class);
        when(session.getTransport("smtp")).thenReturn(transport);
        doThrow(new AddressException("Invalid recipient")).when(transport).sendMessage(any(Message.class), any(Address[].class));

        // Configuração do serviço de email
        EmailService emailService = new EmailService(session, "username", "password");

        // Envio do email deve lançar exceção devido ao destinatário inválido
        emailService.sendEmail("destinatario@example.com", "Assunto", "Corpo");
    }
}

Neste exemplo, utilizamos o Mockito para criar mocks das dependências externas, como a sessão SMTP e o transporte. Isso nos permite simular diferentes cenários, como envio de email bem-sucedido, falha na autenticação SMTP e destinatário inválido, e garantir que o serviço de email se comporte corretamente em cada caso.

import static org.junit.Assert.*;

    }

Aqui está a implementação dos testes unitários para a classe ProductManager utilizando o JUnit e Mockito:

java

import static org.junit.Assert.*;
import static org.mockito.Mockito.*;

import java.util.ArrayList;
import java.util.List;

import org.junit.Test;

public class ProductManagerTest {

    @Test
    public void testAddProduct() {
        // Configuração do mock do repositório de produtos
        ProductRepository productRepository = mock(ProductRepository.class);
        ProductManager productManager = new ProductManager(productRepository);

        // Configuração do produto a ser adicionado
        String name = "Product";
        String description = "Product Description";
        double price = 10.0;

        // Simulação da adição do produto
        when(productRepository.addProduct(name, description, price)).thenReturn(1); // ID do produto adicionado

        // Adição do produto
        int productId = productManager.addProduct(name, description, price);

        // Verificação se o produto foi corretamente adicionado
        assertEquals(1, productId);
        verify(productRepository, times(1)).addProduct(name, description, price);
    }

    @Test(expected = IllegalArgumentException.class)
    public void testAddProductMissingName() {
        ProductRepository productRepository = mock(ProductRepository.class);
        ProductManager productManager = new ProductManager(productRepository);

        // Tentativa de adicionar um produto sem nome
        productManager.addProduct(null, "Description", 10.0);
    }

    @Test
    public void testRemoveProduct() {
        ProductRepository productRepository = mock(ProductRepository.class);
        ProductManager productManager = new ProductManager(productRepository);

        int productId = 1; // ID de um produto existente

        // Simulação da remoção do produto
        when(productRepository.removeProduct(productId)).thenReturn(true);

        // Remoção do produto
        boolean result = productManager.removeProduct(productId);

        // Verificação se o produto foi corretamente removido
        assertTrue(result);
        verify(productRepository, times(1)).removeProduct(productId);
    }

    @Test
    public void testUpdateProduct() {
        ProductRepository productRepository = mock(ProductRepository.class);
        ProductManager productManager = new ProductManager(productRepository);

        int productId = 1; // ID de um produto existente
        String name = "Updated Product";
        String description = "Updated Description";
        double price = 20.0;

        // Simulação da atualização do produto
        when(productRepository.updateProduct(productId, name, description, price)).thenReturn(true);

        // Atualização do produto
        boolean result = productManager.updateProduct(productId, name, description, price);

        // Verificação se o produto foi corretamente atualizado
        assertTrue(result);
        verify(productRepository, times(1)).updateProduct(productId, name, description, price);
    }

    @Test
    public void testListProducts() {
        ProductRepository productRepository = mock(ProductRepository.class);
        ProductManager productManager = new ProductManager(productRepository);

        // Lista simulada de produtos
        List<Product> productList = new ArrayList<>();
        productList.add(new Product(1, "Product1", "Description1", 10.0));
        productList.add(new Product(2, "Product2", "Description2", 20.0));

        // Simulação da listagem de produtos
        when(productRepository.getAllProducts()).thenReturn(productList);

        // Listagem de produtos
        List<Product> retrievedProducts = productManager.listProducts();

        // Verificação se a lista de produtos é retornada corretamente
        assertEquals(productList.size(), retrievedProducts.size());
        verify(productRepository, times(1)).getAllProducts();
    }
}

Neste exemplo, os mocks do Mockito são usados para simular o comportamento do repositório de produtos, permitindo testar isoladamente a lógica da classe ProductManager. Cada método é testado para diferentes cenários e verificações são realizadas para garantir que a interação com o repositório ocorra como esperado.
